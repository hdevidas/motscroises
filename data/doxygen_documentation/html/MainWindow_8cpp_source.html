167void MainWindow::initializeDisplay(unsigned heightOfGrid,unsigned widthOfGrid, std::vector<std::string> grid, std::vector<Word> words, unsigned remainingHelpPoints){
168
169 _showGrid = new ShowGrid(this, widthOfGrid, heightOfGrid, _controller); // creates the graphical grid
170 _showNumber=new ShowNumber(this, widthOfGrid, heightOfGrid);
171 char** numbers = (char **) malloc (heightOfGrid * sizeof(char *));
172 for (unsigned i = 0 ; i < heightOfGrid ; i++) {
173 numbers[i]=(char *) malloc (widthOfGrid * sizeof(char ));
174 for(unsigned j = 0 ; j < widthOfGrid ; j++)
175 numbers[i][j] = '_';
176 }
177
178 std::vector<Word> wordsH;
179 std::vector<Word> wordsV;
180
181 for (unsigned i=0; i<words.size(); i++){
182 Word word =words.at(i);
183 numbers[word.pos.y][word.pos.x] = word.number;
184 if(word.horizontal){
185 wordsH.push_back(word);
186 }
187 else{
188 wordsV.push_back(word);
189 }
190 }
191 _defTable=new DefTable(this,wordsH,wordsV);
192
193 _showNumber->setNumbersInGrid(numbers);
194
195 for (unsigned i = 0 ; i < heightOfGrid ; i++) {
196 free(numbers[i]);
197 }
198 free(numbers);
199
200 this->refreshGrid(grid);
201 this->refreshHelpPoints(remainingHelpPoints);
202
203 //for graphical display
204 ui->gridTable->setModel(_showGrid);
205 ui->gridNumber->setModel(_showNumber);
206 ui->tableDef->setModel(_defTable);
207 ui->gridTable->horizontalHeader()->setSectionResizeMode(QHeaderView::Stretch);
