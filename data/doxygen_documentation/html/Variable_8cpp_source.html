58int Variable::intersectWith(Variable* variable){
59 // Cannot intersect with a variable with the same orientation
60 if(this->_horizontal == variable->_horizontal) {
61 return -1;
62 }
63 int xThis = this->_pos.x;
64 int yThis = this->_pos.y;
65
66 for(int i = 0; i < this->_size; i++) { // Loop through the pos of variable 'this'
67 int xVar = variable->getPos().x;
68 int yVar = variable->getPos().y;
69
70 for(int j = 0; j < variable->getSize(); j++) { // Loop through the pos of the variable we want to check if it intersects with
71 if(xThis == xVar && yThis == yVar){ // It intersects
72 return i;
73 }
74 // Update pos letter to check (variable)
75 if(variable->isHorizontal()) { xVar++; }
76 else { yVar++; }
77 }
78 // Update pos letter to check (this)
79 if(this->_horizontal) { xThis++; }
80 else { yThis++; }
81 }
82 return -1;
83}
