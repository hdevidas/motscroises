242int Crossword::revealWord(unsigned number, bool horizontal){
243 int r = -1;
244 if(_helpPoints - REVEAL_WORD >= 0){
245 _helpPoints -= REVEAL_WORD;
246 r = 1;
247 Word wordToCheck = getWord(number, horizontal);
248 int inc;
249 wordToCheck.horizontal ? inc = wordToCheck.pos.x : inc = wordToCheck.pos.y;
250 for(int i = inc; i <= inc + wordToCheck.word.size()-1; i++){
251 wordToCheck.horizontal ? _currentGrid[wordToCheck.pos.y][i] = _solutionGrid[wordToCheck.pos.y][i] : _currentGrid[i][wordToCheck.pos.x] = _solutionGrid[i][wordToCheck.pos.x];
252 }
253 }
254 return r;
255}
