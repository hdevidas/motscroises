135 for (unsigned j = 0; j < _pattern->getWidth(); j++){

146 p.x = posj;

149 _listVariables[_nbVariables++] = var;
150 variableLength = 0;
151 }
152 else if (lastChar == '_' && currentChar == '#'){ // End of variable (variable not added)
153 variableLength = 0;
154 }
155 }
156 if (currentChar == '_' && variableLength >= database->getLengthMin()){
157 p.x = posj;

225 for (int j = i + 1; j < this->_nbVariables; j++) {
226 pos = this->_listVariables[i]->intersectWith(this->_listVariables[j]);
227 // If it intersects : add mutually the variables and their respective position
228 if(pos != -1){
229 int nPos = _listVariables[j]->intersectWith(this->_listVariables[i]);
230 this->_listVariables[i]->addNeighboors(pos,this->_listVariables[j]);
231 this->_listVariables[j]->addNeighboors(nPos,this->_listVariables[i]);

247 indexMin = i;

269bool Solver::solve() {
270 bool solvable = true;
271
272 std::cout << "Solving the grid..." << std::endl;
273 auto t1 = high_resolution_clock::now(); //to record the duration time
274
275 Variable** fillingOrder = (Variable **) malloc(_nbVariables * sizeof(Variable*)); //Initializes a Variable array to save the order in which the variables have been filled.
276
277 unsigned varIndex = 0; // Each index represents a Variable, when a word inside a variable has been set, index increments
278 unsigned btLevel = 0; // BackTracking level
279 try{
280 while(varIndex < _nbVariables) { //while all the grid is not filled
281 Variable * var = chooseVariable(); //select the variable (the most difficult to insert)
282 bool attributed = true; //to know if the word we'll select is already in the grid
283 int wordIndex = -1;
284
285 while(true){
286 wordIndex = var->getDomain()->findWord(); //gets an index word from the domain variable
287 if(wordIndex == -1){break;}
288 attributed = _attributedWords->isAttributed(var->getSize(),wordIndex); // We look for a word that is not already in the grid
289 if(!attributed){break;}
290 }
291
292 if(wordIndex!=-1) { // If we managed to assign this variable
293 var->setWord(wordIndex); // Add the word to the Variable
294 _attributedWords->addAttributedWord(var->getSize(),wordIndex); // The word is in the grid and will not be allowed to be placed again
295 var->updateNeighbours(); // We update the neighboors's domain of the var that has been placed.
296 fillingOrder[varIndex] = var;
297 varIndex++; // word has been added
298 btLevel = 0;
299 } else { // We can't assign the variable --> we go back and hope to find a possible combination
300
301 // if we can't assign the first variable anymore, it mean the grid is not resoluble
302 if(varIndex == 0){
303 solvable = false;
304 break;
305 }
306
307 varIndex --; //Get the previous variable
308 btLevel++;
309
310 //if the previous variable has blocked a variable, we signal that we must no longer apply the same constraint : in order not to test words for nothing
311 if(var->intersectWith(fillingOrder[varIndex]) != -1 && btLevel == 1)
312 var->conflict(fillingOrder[varIndex]);
313
314 _attributedWords->removeAttributedWord(fillingOrder[varIndex]->getSize(),fillingOrder[varIndex]->getWordIndex());
315 fillingOrder[varIndex]->reset(); //Reseting the previous variable : the next word in his domain will be choose next iteration
316 }
317 this->printGrid();
318
319 }
320 }catch(LexiconException e){
321 std::cerr << e.what() << std::endl;
322 throw SolverLexiconException();
323 }
324 auto t2 = high_resolution_clock::now(); //to record duration time
325
326 /* Getting number of milliseconds as a double. */
327 duration<double, std::milli> ms_double = t2 - t1;
328
329 if(solvable)
330 std::cout << "Grid completed in : " << ms_double.count() << "ms\n";
331
332 free(fillingOrder);
333
334 updateGrid();
335 return solvable;
336}

343std::vector<Word> Solver::getWordVector(){
344 std::vector<Word> words;
345 unsigned number;
346 std::string associateClue, word;
347 bool horizontal;
348 Position pos;
349 int nextNumber = 1;

353
354 pos = _listVariables[i]->getPos();
355 word = _listVariables[i]->getWord();

360 samePos = j;

365 samePos = -1;
366 }
367 else{

369 nextNumber++;
